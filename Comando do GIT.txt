git status// mostra os arquivos que estão no controle e os que não estão
git add . //adiciona o arquivo ao controle de versão

git config --global user.email ""
git config --global user.name "" // configura o usuario do git

git commit -m "mensagem do commit" // adiciona os arquivos no github com o nome da mensagem que você escreveu

git push --set-upstream origin master // adiciona os arquivos no repositorio Master

git remote add origin ( cola o link do repositorio do github )

git reflog // exibi o historico de versões

git reset --hard (digite o codigo do historico )

git branch // exibi as branch 

git branch staging // cria uma nova branch
git checkout -b nome-da-nova-branch master//cria uma nova branch com os dados da branch que você selecionou

git checkout staging // altera para a branch que você selecionou

git merge staging // comando para unir branch Ex: uni as alterações da branch staging para a branch master

git pull // atualiza para a ultima versão das branch ( puxa se caso tiver alteração da maquina servidor )
git pull request// forma mais utilizada, apos fazer a alteração dentro do github você faz a solicitação e o Sr.Rugles aceita ou não que seja feita a merge da sua branch com a branch principal

touch.gitignore // cria um arquivo que será ignorado na hora de subi os arquivos para o github
